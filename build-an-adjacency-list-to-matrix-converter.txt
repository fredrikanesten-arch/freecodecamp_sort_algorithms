def adjacency_list_to_matrix(adj_list):
    
    # 1. Determine size
    n = len(adj_list)

    # 2. Create n x n matrix of zeros
    matrix = [[0 for _ in range(n)] for _ in range(n)]

    # 3. For each node â†’ for each neighbor
    #    set matrix[node][neighbor] = 1
    for node in adj_list:
        for neighbor in adj_list[node]:
            matrix[node][neighbor] = 1

    # 4. Print each row
    for row in matrix:
        print(row)
    
    # 5. Return matrix
    return matrix
