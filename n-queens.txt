def dfs_n_queens(n):
    if n < 1:
        return []

    solutions = []

    def dfs(row, path):
        # If we have placed queens in all rows
        if row == n:
            solutions.append(path[:])  # copy the current path
            return

        for col in range(n):
            # Check if placing at (row, col) is safe
            safe = True
            for r, c in enumerate(path):
                if c == col or abs(row - r) == abs(col - c):
                    safe = False
                    break

            if safe:
                path.append(col)
                dfs(row + 1, path)
                path.pop()  # backtrack

    dfs(0, [])
    return solutions
