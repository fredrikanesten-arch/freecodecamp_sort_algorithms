def verify_card_number(card_number):
    account_number = card_number.replace(" ", "").replace("-", "")
    total = 0

    reversed_digits = account_number[::-1]

    for i, digit in enumerate(reversed_digits):
        n = int(digit)

        if i % 2 == 1:
            n *= 2
            if n > 9:
                n -= 9

        total += n

    if total % 10 == 0:
        return 'VALID!'
    else:
        return 'INVALID!'
